# the sub body is taken from PDD03
.sub main :main
    .local pmc x, y
    foo(x :flat)
    foo(x, 'y' => y)
    foo(x, y :named('y'))
    foo(x :flat :named)
    foo(a, b, c :flat, 'x' => 3, 'y' => 4, z :flat :named('z'))

    x = foo()                       # single result
    (i, j :optional, ar :slurpy, value :named('key') ) = foo()
.end

.sub foo
    .return (i, ar :flat, value :named('key') )
.end
